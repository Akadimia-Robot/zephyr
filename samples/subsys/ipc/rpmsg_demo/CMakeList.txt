cmake_minimum_required(VERSION 3.20.0)

find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})
project(rpmsg_demo)

# METAL_MAX_DEVICE_REGIONS is used to give the number of memory regions shared between processors.
# By default only one region is defined for vdev buffer.
# The METAL_MAX_DEVICE_REGIONS has to be redifined to add a second region for the resource table.
zephyr_compile_definitions(METAL_MAX_DEVICE_REGIONS=2)

target_include_directories(app
    PRIVATE
        src
)

target_sources(app
    PRIVATE
        src/main.cpp
        src/rsc_table.cpp
)
